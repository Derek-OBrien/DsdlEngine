39  std::string temp = FileIO::getInstance()->getWritablePath() + texturePath;
40  //Store in map for loading
41  auto it = m_TextureMap.find(temp);
42 
43  SDL_Texture* newTexture = NULL;
44  SDL_Surface* loadedSurface = NULL;
45 
46  if (it == m_TextureMap.end()){
47  //Load image at specified path
48  loadedSurface = IMG_Load(temp.c_str());
49 
50 
51  if (loadedSurface == NULL)
52  SDL_Log("SDL_image Error : %s ", IMG_GetError());
53  else{
54  //Color key image
55  SDL_SetColorKey(loadedSurface, SDL_TRUE, SDL_MapRGB(loadedSurface->format, 0, 0xFF, 0xFF));
56 
57  //Create texture from surface pixels
58  newTexture = SDL_CreateTextureFromSurface(r, loadedSurface);
59  if (newTexture == NULL){
60  SDL_Log("SDL_CreateTextureFromSurface Error : %s ",IMG_GetError());
61  }
62  else{
63  //Get image dimensions
64  m_iWidth = loadedSurface->w;
65  m_iHeight = loadedSurface->h;
66  }
67  //Get rid of old loaded surface
68  SDL_FreeSurface(loadedSurface);
69  }
70  //Add to map
71  m_TextureMap[temp] = newTexture;
72  }
73  else{
74  newTexture = it->second;
75  }
76  //Return success
77  m_Texture = newTexture;
